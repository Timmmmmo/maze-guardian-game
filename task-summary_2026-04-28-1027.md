  ctx.save();
  ctx.beginPath();
  ctx.arc(x, y - size*0.1, imgSize/2, 0, Math.PI * 2);
  ctx.clip();
  ctx.drawImage(assets[imgAsset], x - imgSize/2, y - imgSize/2 - size*0.1, imgSize, imgSize);
  ctx.restore();
  return;
}

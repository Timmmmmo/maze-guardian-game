  ctx.drawImage(img, x - img.width/2, y - img.height/2);

### 4.2 攻击轨迹系统

```javascript
// 箭矢轨迹
function drawArrowTrail(from, to, progress) {
  ctx.beginPath();
  ctx.moveTo(from.x, from.y);
  ctx.lineTo(
    from.x + (to.x - from.x) * progress,
    from.y + (to.y - from.y) * progress
  );
  ctx.strokeStyle = '#FFB347';
  ctx.lineWidth = 2;
  ctx.stroke();
}

// 爆炸特效
function drawExplosion(x, y, radius, progress) {
  ctx.beginPath();
  ctx.arc(x, y, radius * progress, 0, Math.PI * 2);
  ctx.fillStyle = `rgba(255, 100, 0, ${1 - progress})`;
  ctx.fill();
}
```
